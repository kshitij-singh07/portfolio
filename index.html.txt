<!-- ── SCRIPTS ────────────────────────────────────────────────── -->
<script>
  /* ── NAV scroll ── */
  const navbar = document.getElementById('navbar');
  window.addEventListener('scroll', () => {
    navbar.classList.toggle('scrolled', window.scrollY > 20);
  });

  /* ── Hamburger ── */
  const ham = document.getElementById('ham');
  const mobileMenu = document.getElementById('mobileMenu');
  ham.addEventListener('click', () => {
    mobileMenu.classList.toggle('open');
  });
  function closeMobile() { mobileMenu.classList.remove('open'); }

  /* ── Terminal typewriter ── */
  const lines = [
    { id: 'tline1', content: '<span class="prompt">~ </span><span class="cmd">whoami</span>' },
    { id: 'tline2', content: '<span class="prompt">→ </span><span class="output">Kshitij Singh</span>&nbsp;<span class="key">// CS Engineer · Builder</span>' },
    { id: 'tline3', content: '<span class="prompt">~ </span><span class="cmd">cat skills.txt</span>&nbsp;<span class="cursor"></span>' },
  ];
  let li = 0;
  function typeLine() {
    if (li >= lines.length) return;
    const el = document.getElementById(lines[li].id);
    el.innerHTML = lines[li].content;
    li++;
    setTimeout(typeLine, li === 1 ? 300 : 500);
  }
  setTimeout(typeLine, 800);

  /* ── Reveal on scroll ── */
  const revealEls = document.querySelectorAll('.reveal');
  const observer = new IntersectionObserver((entries) => {
    entries.forEach((e, i) => {
      if (e.isIntersecting) {
        // stagger siblings
        const siblings = [...e.target.parentElement.querySelectorAll('.reveal')];
        const idx = siblings.indexOf(e.target);
        setTimeout(() => {
          e.target.classList.add('visible');
        }, idx * 80);
        observer.unobserve(e.target);
      }
    });
  }, { threshold: 0.12 });
  revealEls.forEach(el => observer.observe(el));

  /* ── Contact form ── */
  document.getElementById('contactForm').addEventListener('submit', function(e) {
    e.preventDefault();
    const btn = this.querySelector('.form-submit');
    btn.textContent = 'Sending...';
    btn.disabled = true;
    setTimeout(() => {
      btn.textContent = 'Send Message →';
      btn.disabled = false;
      document.getElementById('formMsg').classList.add('success');
      this.reset();
      setTimeout(() => {
        document.getElementById('formMsg').classList.remove('success');
      }, 5000);
    }, 1200);
  });
</script>
